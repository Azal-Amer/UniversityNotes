Part 2: Testing (20 points)

public class ImageProcessorTest {
    private static void printArray(int[][] arr) {
        for(int[] row : arr) {
            for(int val : row) {
                System.out.printf("%4d", val);
            }
            System.out.println();
        }
        System.out.println();
    }
    
    private static boolean arrayEquals(int[][] arr1, int[][] arr2) {
        if(arr1.length != arr2.length || arr1[0].length != arr2[0].length) {
            return false;
        }
        for(int i = 0; i < arr1.length; i++) {
            for(int j = 0; j < arr1[0].length; j++) {
                if(arr1[i][j] != arr2[i][j]) return false;
            }
        }
        return true;
    }
    
    public static void main(String[] args) {
        int testsPassed = 0;
        int totalTests = 0;
        
        // Test 1: Average Brightness
        {
            totalTests++;
            int[][] testImage = {
                {100, 200},
                {50, 150}
            };
            ImageProcessor proc = new ImageProcessor(testImage);
            double avg = proc.getAverageBrightness();
            System.out.println("Test 1: Average Brightness");
            System.out.println("Expected: 125.0");
            System.out.println("Got: " + avg);
            if(Math.abs(avg - 125.0) < 0.001) {
                System.out.println("PASSED");
                testsPassed++;
            } else {
                System.out.println("FAILED");
            }
            System.out.println();
        }
        
        // Test 2: Invert
        {
            totalTests++;
            int[][] testImage = {
                {100, 200},
                {50, 150}
            };
            int[][] expected = {
                {155, 55},
                {205, 105}
            };
            ImageProcessor proc = new ImageProcessor(testImage);
            System.out.println("Test 2: Invert");
            System.out.println("Original:");
            printArray(testImage);
            proc.invert();
            System.out.println("After invert:");
            printArray(proc.getImage());
            System.out.println("Expected:");
            printArray(expected);
            if(arrayEquals(proc.getImage(), expected)) {
                System.out.println("PASSED");
                testsPassed++;
            } else {
                System.out.println("FAILED");
            }
            System.out.println();
        }
        
        // Test 3: Flip Horizontal
        {
            totalTests++;
            int[][] testImage = {
                {1, 2, 3},
                {4, 5, 6}
            };
            int[][] expected = {
                {3, 2, 1},
                {6, 5, 4}
            };
            ImageProcessor proc = new ImageProcessor(testImage);
            System.out.println("Test 3: Flip Horizontal");
            System.out.println("Original:");
            printArray(testImage);
            proc.flipHorizontal();
            System.out.println("After flip:");
            printArray(proc.getImage());
            System.out.println("Expected:");
            printArray(expected);
            if(arrayEquals(proc.getImage(), expected)) {
                System.out.println("PASSED");
                testsPassed++;
            } else {
                System.out.println("FAILED");
            }
            System.out.println();
        }
        
        // Test 4: Checkerboard
        {
            totalTests++;
            int[][] expected = {
                {255, 0, 255},
                {0, 255, 0},
                {255, 0, 255}
            };
            System.out.println("Test 4: Checkerboard");
            int[][] board = ImageProcessor.createCheckerboard(3);
            System.out.println("Created board:");
            printArray(board);
            System.out.println("Expected:");
            printArray(expected);
            if(arrayEquals(board, expected)) {
                System.out.println("PASSED");
                testsPassed++;
            } else {
                System.out.println("FAILED");
            }
        }
        
        // Final Score
        System.out.println("\nTests passed: " + testsPassed + "/" + totalTests);
        double score = (testsPassed * 100.0) / totalTests;
        System.out.println("Score: " + score + "%");
    }
}

Overloading

public class OverloadExample {
    public void method(int i) {
        System.out.println("Method with int");
    }
    

    
    public static void main(String[] args) {
        OverloadExample ex = new OverloadExample();
        
        ex.method(5);        // Calls method(int)
        ex.method(5L);       // Calls method(long)
        ex.method(Integer.valueOf(5)); // Calls method(Integer)
        ex.method(5, 6, 7);  // Calls method(int...)
        ex.method((short)5); // Calls method(int) - widening
    }
}

overriding

class Animal {
    public void makeSound() {
        System.out.println("The animal makes a sound");
    }
}

class Dog extends Animal {
    @Override
    public void makeSound() {
        System.out.println("The dog barks");
    }
}

import java.util.Scanner;  
  
public class Main {  
    public static void main(String[] args) {  
        String c = "\uD83D\uDCA9";  
        Scanner input = new Scanner(System.in);  
        int x= input.nextInt();  
        for(int i = 0;i<x;i++){  
            for(int j = x;j>i;j--){  
                System.out.print(c);  
            }  
            System.out.println();  
        }  
    }  
}

import math

# Constants
g = 9.8  # acceleration due to gravity, in m/s^2

def question_2(mu_k, theta_deg, mass):
    theta = math.radians(theta_deg)
    # Correct answer
    correct = g * (math.sin(theta) - mu_k * math.cos(theta))
    
    # Wrong answers
    no_friction = g * math.sin(theta)
    added_friction = g * (math.sin(theta) + mu_k * math.cos(theta))
    wrong_trig = g * (math.cos(theta) - mu_k * math.sin(theta))
    
    return {
        "Wrong": [round(no_friction, 2), round(added_friction, 2), round(wrong_trig, 2)],
        "Correct": [round(correct, 2)]
    }

def question_5(m1, m2, theta_deg):
    theta = math.radians(theta_deg)
    # Correct tension formula
    correct_tension = (m2 * g) - (m1 * g * math.sin(theta))
    
    # Wrong answers
    no_angle_component = m2 * g - m1 * g
    ignore_second_mass = m1 * g * math.sin(theta)
    wrong_accel = (m2 * g + m1 * g * math.cos(theta))
    
    return {
        "Wrong": [round(no_angle_component, 2), round(ignore_second_mass, 2), round(wrong_accel, 2)],
        "Correct": [round(correct_tension, 2)]
    }

def question_6(mass, theta_deg, mu_k):
    theta = math.radians(theta_deg)
    # Normal force calculation
    normal_force = mass * g * math.cos(theta)
    
    # Correct friction force
    correct_friction = mu_k * normal_force
    
    # Wrong answers
    wrong_trig = mu_k * mass * g * math.sin(theta)
    add_weight = mu_k * (mass * g + normal_force)
    subtract_weight = mu_k * (normal_force - mass * g)
    
    return {
        "Wrong": [round(wrong_trig, 2), round(add_weight, 2), round(subtract_weight, 2)],
        "Correct": [round(correct_friction, 2)]
    }

def question_7(m1, m2, mu_k):
    # Correct acceleration
    correct_accel = (m2 * g - mu_k * m1 * g) / (m1 + m2)
    
    # Wrong answers
    no_friction = (m2 * g) / (m1 + m2)
    wrong_mu = (m2 * g - (mu_k + 0.1) * m1 * g) / (m1 + m2)
    wrong_total_force = (m2 * g + mu_k * m1 * g) / (m1 + m2)
    
    return {
        "Wrong": [round(no_friction, 2), round(wrong_mu, 2), round(wrong_total_force, 2)],
        "Correct": [round(correct_accel, 2)]
    }

# Example calls to generate answers for each numerical question
Q2_answers = question_2(0.2, 30, 2)  # mu_k = 0.2, theta = 30°, mass = 2 kg
Q5_answers = question_5(2, 3, 30)  # m1 = 2 kg, m2 = 3 kg, theta = 30°
Q6_answers = question_6(5, 37, 0.25)  # mass = 5 kg, theta = 37°, mu_k = 0.25
Q7_answers = question_7(2, 3, 0.1)  # m1 = 2 kg, m2 = 3 kg, mu_k = 0.1

# Output the answers
answers = {
    "Q2": Q2_answers,
    "Q5": Q5_answers,
    "Q6": Q6_answers,
    "Q7": Q7_answers
}

print(answers)
